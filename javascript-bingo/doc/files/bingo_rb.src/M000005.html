def draw_card(words)
  page_width = bounds.width
  left_edge = (bounds.width - CELL_WIDTH*5)/2
  bounding_box [left_edge,0], :width => CELL_WIDTH*5 do
    start = [600,50]
    %w(J a v a S c r i p t).each_with_index do |letter, index|
      #draw_box(letter, (5*CELL_WIDTH*index)/11, -100, true, {:size => 20, :style => :bold})
    end
    
  bounding_box [0, 700], :width => 5*CELL_WIDTH do
    text "JavaScript Bingo", :align => :center, :size => 24, :style => :bold, :character_spacing => 4
    stroke_horizontal_rule
  end
  
    %w(B I N G O).each_with_index do |letter, index|
      draw_box(letter, CELL_WIDTH*index, -60, false, {:size => 20, :style => :bold})
    end
    (0..4).each do |row|
      (0..4).each do |col|
        word = get_word(words,row,col)
        x = CELL_WIDTH*row
        y = CELL_HEIGHT*col
        draw_box(word,x,y)
      end
    end
    column_box([0,cursor-100], :columns => 3, :align => :center, :width => bounds.width) do
      text WORDS.join("\n")
    end
    draw_footer
    
  end
end
